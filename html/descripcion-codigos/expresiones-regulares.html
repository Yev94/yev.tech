Ejemplos de Expresiones regulares

        
const EXP = {
    user: /^[a-zA-Z0-9\_\-]{4,16}$/, //Letras, números, guiones y guiones bajos
    name: /^(\b[A-zÀ-ÿ]*\b[ ]?)+$/, //Letras con acento, números y espacio
    password: /^.{4,12}$/, //De 4 a 9 dígitos
    email: /[a-z]+([\.-]?\w+)*@[a-z]+([\.-]?\w+)*\.[a-z]{2,}/,
    tel: /^((\+|00)?)\d{11}|\d{9}$/, //9 dígitos o {+/00 y 11 dígitos} para internacional
}
