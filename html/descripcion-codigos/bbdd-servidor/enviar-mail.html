Enviar mail a través del servidor

En PHP

            
//Recibimos la info a través del correspondiente formulario
//Para enviar los correos hay que configurar el smtp del servidor y los accesos de gmail
$email = $_POST['email'];
$subject = $_POST['subject'];
$message = $_POST['message'];

$headers = "MIME-Version: 1.0" . "\r\n";
$headers .= "Content-type:text/html;charset=UTF-8" . "\r\n";
$headers .= 'From: Prueba Yev < ejemplo@gmail.com >' . "\r\n";

$exito = mail($email, $subject, $message, $headers);

if ($exito) {
    echo "Mensaje enviado";
} else {
    echo "Mensaje no enviado";
}
