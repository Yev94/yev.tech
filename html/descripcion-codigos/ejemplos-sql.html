De Wordpress sacar todos los títulos y las URL de las imágenes destacadas

        
SELECT p.ID, p.post_title, pm.meta_value AS 'featured_image_id', pm2.meta_value AS 'featured_image_url' 
  FROM wp_posts p LEFT JOIN wp_postmeta pm ON p.ID = pm.post_id AND pm.meta_key = '_thumbnail_id' 
  LEFT JOIN wp_posts p2 ON pm.meta_value = p2.ID 
  LEFT JOIN wp_postmeta pm2 ON pm.meta_value = pm2.post_id AND pm2.meta_key = '_wp_attached_file' 
  WHERE p.post_type = 'post' AND p.post_status = 'publish';
