//Podemos crear un iterador que nos devuelva los primeros n elementos de una serie de fibonnaci
function* fibonnaci() {
    let a = 0;
    let b = 1;
    while (true) {
        yield a;
        [a, b] = [b, a + b];
    }

}
//Ãštil si no sabemos las veces que vamos a iterar
let [a, b, c, d, e, f, g, h, i, j] = fibonnaci();
console.log(a, b, c, d, e, f, g, h, i, j);

//Otra forma de crear el fibonnaci con el bucle for
function fibonnaci2(n) {
    let a = 0;
    let b = 1;
    for (let i = 0; i < n; i++) {
        [a, b] = [b, a + b];
    }
    return a;
}
console.log(fibonnaci2(10));


//Generar el fibonnaci con recursividad
function fibonnaci3(n) {
    if(n < 2) return n;
    return fibonnaci3(n - 1) + fibonnaci3(n - 2);
}
console.log(fibonnaci3(10));
