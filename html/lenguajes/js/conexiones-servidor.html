Mediante XML HTTP Request

Conectar con GET sin llevar info adicional

        
const xhr = new XMLHttpRequest(); 
const url = './holamundo.txt'; 

xhr.addEventListener('readystatechange', function(){ 
  if(this.readyState !== 4) return;
  if (this.status >= 200 && this.status < 300) { 
    let contenido = this.responseText;
} 
})
xhr.open('GET', url, true); 
xhr.send();

Mediante Fetch

Conectar con GET sin llevar info adicional

          
const url = './holamundo.txt';
fetch(url)
.then(function(response) {
  if(response.ok) {
    return response.text();
  } else {
    throw new Error(response.status);
  }
})
.then(function(texto) {
  console.log(texto);
})
.catch(function(error) {
  console.log(error);
});

Para conectar con POST enviando info adicional al server antes

          
const url = './holamundo.txt';
fetch(url, {
  method: 'POST',
  body: loQueQueramosEnviar
})
.then(function(response) {
  if(response.ok) {
    return response.text();
  } else {
    throw new Error(response.status);
  }
})
.then(function(texto) {
  console.log(texto);
})
.catch(function(error) {
  console.log(error);
});
