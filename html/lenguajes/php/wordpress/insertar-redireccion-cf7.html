Insertar Redirección Contact Form 7

Insertamos en un hook del footer un código JS para que cambie a la página deseada

      
// ------------- Inicio CF7 Redireccion ----------------------
function yev_redirection_form() {?>
<script>
// Nos ponemos a la escucha del envío del formulario
document.addEventListener( 'wpcf7mailsent', function( event ) {
  // En cuanto se active redireccionamos a la página deseada
    location = 'http://niveldecalidad.com/gracias/';
}, false );
</script>
<?php
}
add_action( 'wp_footer', 'yev_redirection_form' ); //Hook del Footer
// -------------- Fin CF7 Redireccion -------------------------
